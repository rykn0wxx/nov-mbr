Getting Start Date and Group

  
function initialDate(obj) {
	var l = obj.length;
	var mindate = new GlideDateTime(obj[0].start);
	var iGroup = '', iDate = '';
	for (var i = 0; i !== l; i++) {
		var curDate = new GlideDateTime(obj[i].start);
		if (mindate.compareTo(curDate) ===  1) {
			mindate = new GlideDateTime(obj[i].start);
			iDate = obj[i].start;
			iGroup = obj[i].groupname;
		}
	}
	return {'start':iDate, 'group': iGroup };
}
